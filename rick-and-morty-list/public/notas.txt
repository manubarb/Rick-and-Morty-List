- função para atualizar os contadores no footer
    async function updateCounts(charCount) {
        try {
            // Atualizar contagem de personagens
            characterCount.textContent = charCount;
            
            // Buscar contagem de localizações
            const locationsResponse = await fetch('https://rickandmortyapi.com/api/location');
            const locationsData = await locationsResponse.json();
            locationCount.textContent = locationsData.info.count;
            
            // Buscar contagem de episódios
            const episodesResponse = await fetch('https://rickandmortyapi.com/api/episode');
            const episodesData = await episodesResponse.json();
            episodeCount.textContent = episodesData.info.count;
        } catch (error) {
            console.error('Error updating counts:', error);
        }
    }

        if (characters.length === 0) {
            cardsContainer.innerHTML = '<p class="text-white">Nenhum personagem encontrado</p>';
            return;
        }

            const card = createCharacterCard(character);
